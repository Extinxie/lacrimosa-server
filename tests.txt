Как сделать простую авторизацию / регистрацию в Elysia
Elysia уже имеет готовый JWT‑плагин (@elysiajs/jwt).
С его помощью можно:

Создать пользователя (записать в память/БД).
Сгенерировать JWT‑токен после входа.
Проверять токен в защищённых роутерах.
Ниже минимальный пример, который работает в любой среде (Node, Bun, Deno …).

// 1️⃣ Установить зависимости
// npm i elysia @elysiajs/jwt t
import { Elysia, t } from 'elysia'
import { jwt } from '@elysiajs/jwt'
// 👤 Временное хранилище пользователей (для демо)
// В реальном проекте используйте БД и хеширование пароля.
type User = { id: number; username: string; password: string }
const users: User[] = []
// 🔑 Секрет, которым подписывается JWT
const JWT_SECRET = 'super‑secret‑key‑change‑me!'
// 2️⃣ Создаём приложение
const app = new Elysia()
  // подключаем плагин JWT
  .use(
    jwt({
      name: 'jwt',          // будет доступен как ctx.jwt
      secret: JWT_SECRET,
      // токен будет храниться в cookie "auth"
      cookie: { name: 'auth', httpOnly: true, sameSite: 'strict' },
    })
  )
  // ── Регистрация ────────────────────────────────────────
  .post(
    '/register',
    ({ body }) => {
      // проверяем, что пользователь ещё не существует
      if (users.find(u => u.username === body.username))
        return { error: 'User already exists' }
      const newUser: User = {
        id: users.length + 1,
        username: body.username,
        password: body.password, // ← в продакшене храните хеш!
      }
      users.push(newUser)
      return { success: true }
    },
    {
      // типы входных данных – авто‑документация + валидация
      body: t.Object({
        username: t.String({ minLength: 3 }),
        password: t.String({ minLength: 6 }),
      }),
    }
  )
  // ── Вход (логин) ───────────────────────────────────────
  .post(
    '/login',
    async ({ body, jwt, set }) => {
      const user = users.find(
        u => u.username === body.username && u.password === body.password
      )
      if (!user) return { error: 'Invalid credentials' }
      // создаём JWT‑токен (можно добавить expiry)
      const token = await jwt.sign({ id: user.id, username: user.username })
      // кладём токен в cookie
      set.cookie.auth = token
      return { success: true }
    },
    {
      body: t.Object({
        username: t.String(),
        password: t.String(),
      }),
    }
  )
  // ── Защищённый роут ─────────────────────────────────────
  .get(
    '/me',
    async ({ jwt }) => {
      // jwt.verify() бросит ошибку, если токен недействителен
      const payload = await jwt.verify()
      // payload = { id: number; username: string }
      return { user: payload }
    },
    {
      // указываем, что роут требует токен
      // если токен не найден/некорректен – будет 401
      beforeHandle: [{ name: 'jwt' }],
    }
  )
  // ── Выход (удалить cookie) ─────────────────────────────────
  .post('/logout', ({ set }) => {
    set.cookie.auth = '' // очистить
    return { success: true }
  })
  .listen(3000)
console.log('🦊 Elysia running on http://localhost:3000')
